// 组件
const components = {
  ...,
  ...W***Bootstrap.components,
};
// 配置
const configs = {
  ...,
  ...W***Bootstrap.config,
};

# Instance.js

实例注册类

export default class Instance {
  libraries = {};
  register(hook, that) {
    this.libraries[hook] = that;
  }
  get(hook) {
    return this.libraries[hook];
  }
}

# options

主要作为组件的基本信息说明。其基本项为：

  • title: 组件名称
  • eventEnums: 预设事件列表，label 事件说明，name 事件名称

# slotName

插槽名称，如果此组件为插槽组件，则必须要配置此属性，默认值为 default。

# component/index.vue

组件模板，例：

<template>
  <el-button class="w-button" v-bind="data.attribs" v-on="events">
    {{ buttonText }}
  </el-button>
</template>

<script>
import componentMixin from "../../../componentMixin";
export default {
  mixins: [componentMixin],
  data() {
    return {
      // 数据绑定
      buttonText: this.data.datas.find((obj) => obj.attribs === "buttonText").defaultValue,
    };
  },
};
</script>

inject: ["config", "currentActive"] config 为页面配置，currentActive 为当前组件配置，都可通过 this 访问，属性配置即对 currentActive 数据进行修改， 属性配置所用表单控件都需使用 ElementUI 扩展控件，即使用 el-input 需换成 elx-input，扩展组件位于 components/ElxComponents/***，数据绑定需从原来的 v-model="data" 更换成 :value.sync="data" 其余属性使用方式和 ElementUI 一致。

平台提供一种样式解决方案，配置按块分割。即根类名必须为 option，其下可通过属性类型分类，创建不同的卡片，其类名为 card，卡片里面也可根据内容不同指定不同的类，即 header、body、footer，一个卡片可拥有多个 body。

# 组件开发

新建一个组件必须创建组件所必须文件，并且更改相关配置项。

创建一个新组件流程如下：

  1. 在 library/components 下创建一个组件文件夹，文件夹名称为 W***，同时文件夹名称即为组件名称，后续配置需要保持一致。
  2. 创建相关文件，如 componet/index.vue、option/index.vue、bootstrap.js、index.js。
  3. 配置组件关联，在 bootstrap.js 中配置组件相关文件关联，配置完成后再在全局引导（library/index.js）中配置引入。
  4. 编写模板文件并且编写配置文件，至此一个新组件开发完成。

# 基座打包

组件开发完成后，需要上传至平台供用户使用，此时则需要进行打包操作。

打包流程如下：

  1. 返回项目根目录。
  2. 执行命令：yarn build 开始打包。
  3. 打包完成，打包生成文件 dist/index.js。至此打包完成。

# 配置资源包

资源包可直接上传至平台。

配置流程如下：

  1. 创建一个资源包文件夹，名称任意，并进入资源包文件夹内。
  2. 创建文件夹 screenshot，此文件夹即为组件缩略图，格式为png 格式，文件名称和组件名称必须相同。
  3. 创建 data.json 文件，该文件即指明该资源包的相关内容，格式如下:
{
  "libraryName": "组件库名称",
  "label": "标签一,标签二",
  "data": [
    {
      "name": "容器",
      "label": "容器,WContainer",
      "componentName": "WContainer",
      "themeVar": [
        {
          "name": "--w-container-border-color",
          "value": "#ffffff",
          "type": "color"
        }
      ]
    }
  ]
}
